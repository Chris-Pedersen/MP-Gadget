178

180  All.UnitTime_in_Megayears = All.UnitTime_in_s / SEC_PER_MEGAYEAR;
181 
182  All.G = GRAVITY / pow(All.UnitLength_in_cm, 3) * All.UnitMass_in_g * pow(All.UnitTime_in_s, 2);
183 
184  All.UnitDensity_in_cgs = All.UnitMass_in_g / pow(All.UnitLength_in_cm, 3);
185  All.UnitPressure_in_cgs = All.UnitMass_in_g / All.UnitLength_in_cm / pow(All.UnitTime_in_s, 2);
186  All.UnitCoolingRate_in_cgs = All.UnitPressure_in_cgs / All.UnitTime_in_s;
187  All.UnitEnergy_in_cgs = All.UnitMass_in_g * pow(All.UnitLength_in_cm, 2) / pow(All.UnitTime_in_s, 2);
188 
189  /* convert some physical input parameters to internal units */
190 
191  All.CP.Hubble = HUBBLE * All.UnitTime_in_s;
192  init_cosmology(&All.CP, All.TimeInit, All.BoxSize * All.UnitLength_in_cm);
193  /*Initialise the hybrid neutrinos, after Omega_nu*/
194  if(All.HybridNeutrinosOn)
195  init_hybrid_nu(&All.CP.ONu.hybnu, All.CP.MNu, All.HybridVcrit, C/1e5, All.HybridNuPartTime, All.CP.ONu.kBtnu);
196 
197  meanweight = 4.0 / (1 + 3 * HYDROGEN_MASSFRAC); /* note: assuming NEUTRAL GAS */
198 
199  All.MinEgySpec = 1 / meanweight * (1.0 / GAMMA_MINUS1) * (BOLTZMANN / PROTONMASS) * All.MinGasTemp;
200  All.MinEgySpec *= All.UnitMass_in_g / All.UnitEnergy_in_cgs;
201 
202 #ifdef SFR
203 
204  All.OverDensThresh =
205  All.CritOverDensity * All.CP.OmegaBaryon * 3 * All.CP.Hubble * All.CP.Hubble / (8 * M_PI * All.G);
206 
207  All.PhysDensThresh = All.CritPhysDensity * PROTONMASS / HYDROGEN_MASSFRAC / All.UnitDensity_in_cgs;
208 
209  All.EgySpecCold = 1 / meanweight * (1.0 / GAMMA_MINUS1) * (BOLTZMANN / PROTONMASS) * All.TempClouds;
210  All.EgySpecCold *= All.UnitMass_in_g / All.UnitEnergy_in_cgs;
211 
212  meanweight = 4 / (8 - 5 * (1 - HYDROGEN_MASSFRAC)); /* note: assuming FULL ionization */
213 
214  All.EgySpecSN = 1 / meanweight * (1.0 / GAMMA_MINUS1) * (BOLTZMANN / PROTONMASS) * All.TempSupernova;
215  All.EgySpecSN *= All.UnitMass_in_g / All.UnitEnergy_in_cgs;
216 
217  if(All.WindOn) {
218  if(HAS(All.WindModel, WIND_FIXED_EFFICIENCY)) {
219  All.WindSpeed = sqrt(2 * All.WindEnergyFraction * All.FactorSN * All.EgySpecSN / (1 - All.FactorSN) / All.WindEfficiency);
220  message(0, "Windspeed: %g\n", All.WindSpeed);
221  } else {

224  }
225  }
226 #endif
227

229  message(0, "G (internal units) = %g\n", All.G);
230  message(0, "UnitLengh_in_cm = %g \n", All.UnitLength_in_cm);
231  message(0, "UnitMass_in_g = %g \n", All.UnitMass_in_g);

235  message(0, "UnitEnergy_in_cgs = %g \n", All.UnitEnergy_in_cgs);
236  message(0, "Photon density OmegaG = %g\n",All.CP.OmegaG);
237  if(!All.MassiveNuLinRespOn)
238  message(0, "Massless Neutrino density OmegaNu0 = %g\n",get_omega_nu(&All.CP.ONu, 1));
239  message(0, "Curvature density OmegaK = %g\n",All.CP.OmegaK);
240  if(All.CP.RadiationOn) {
