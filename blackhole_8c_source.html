244 static void
245 blackhole_accretion_postprocess(int i, TreeWalk * tw)
246 {
247  if(BHP(i).Density > 0)
248  {
249  BHP(i).Entropy /= BHP(i).Density;
250  BHP(i).Pressure /= BHP(i).Density;
251 
252  BHP(i).SurroundingGasVel[0] /= BHP(i).Density;
253  BHP(i).SurroundingGasVel[1] /= BHP(i).Density;
254  BHP(i).SurroundingGasVel[2] /= BHP(i).Density;
255  }
256  double mdot = 0; /* if no accretion model is enabled, we have mdot=0 */
257 
258  double rho = BHP(i).Density;
259  double bhvel = sqrt(pow(P[i].Vel[0] - BHP(i).SurroundingGasVel[0], 2) +
260  pow(P[i].Vel[1] - BHP(i).SurroundingGasVel[1], 2) +
261  pow(P[i].Vel[2] - BHP(i).SurroundingGasVel[2], 2));
262 
263  bhvel /= All.cf.a;
264  double rho_proper = rho * All.cf.a3inv;
265 
266  double soundspeed = blackhole_soundspeed(BHP(i).Entropy, BHP(i).Pressure, rho);
267 
268  /* Note: we take here a radiative efficiency of 0.1 for Eddington accretion */
269  double meddington = (4 * M_PI * GRAVITY * C * PROTONMASS / (0.1 * C * C * THOMPSON)) * BHP(i).Mass
270  * All.UnitTime_in_s;
271 
272  double norm = pow((pow(soundspeed, 2) + pow(bhvel, 2)), 1.5);
273 
274  if(norm > 0)
275  mdot = 4. * M_PI * All.BlackHoleAccretionFactor * All.G * All.G *
276  BHP(i).Mass * BHP(i).Mass * rho_proper / norm;
277  else
278  mdot = 0;
279 
280  if(All.BlackHoleEddingtonFactor > 0.0 &&
281  mdot > All.BlackHoleEddingtonFactor * meddington) {
282  mdot = All.BlackHoleEddingtonFactor * meddington;
283  }
284  BHP(i).Mdot = mdot;
285 
286  double dtime = get_dloga_for_bin(P[i].TimeBin) / All.cf.hubble;
287 
288  BHP(i).Mass += BHP(i).Mdot * dtime;
289 }
