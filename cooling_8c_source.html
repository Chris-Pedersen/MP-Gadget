120 {
121  const double rhoc = 3.0 * pow(All.CP.HubbleParam*HUBBLE,2.0) /(8.0*M_PI*GRAVITY);
122  const double rho = PROTONMASS * nHcgs / XH;
123  const double overden = rho/(All.CP.OmegaBaryon * rhoc * pow(All.Time,-3.0));

125  return All.HeliumHeatAmp*pow(All.HeliumHeatThresh, All.HeliumHeatExp);

133 double DoCooling(double u_old, double rho, double dt, struct UVBG * uvbg, double *ne_guess, double Z)
134 {
135  if(CoolingNoPrimordial) return 0;
136 
137  double u, du;
138  double u_lower, u_upper;
139  double ratefact;
140  double LambdaNet;
141  int iter = 0;
142 
143  rho *= All.UnitDensity_in_cgs * All.CP.HubbleParam * All.CP.HubbleParam; /* convert to physical cgs units */
144  u_old *= All.UnitPressure_in_cgs / All.UnitDensity_in_cgs;
145  dt *= All.UnitTime_in_s / All.CP.HubbleParam;
146 
147  double nHcgs = XH * rho / PROTONMASS; /* hydrogen number dens in cgs units */
148  ratefact = nHcgs * nHcgs / rho;
149 
150  u = u_old;
151  u_lower = u;
152  u_upper = u;
153 
154  LambdaNet = CoolingRateFromU(u, nHcgs, uvbg, ne_guess, Z);
155 
156  /* bracketing */
157 
158  if(u - u_old - ratefact * LambdaNet * dt < 0) /* heating */
159  {
160  u_upper *= sqrt(1.1);
161  u_lower /= sqrt(1.1);
162  while(u_upper - u_old - ratefact * CoolingRateFromU(u_upper, nHcgs, uvbg, ne_guess, Z) * dt < 0)
163  {
164  u_upper *= 1.1;
165  u_lower *= 1.1;
166  }
167 
168  }
169 
170  if(u - u_old - ratefact * LambdaNet * dt > 0)
171  {
172  u_lower /= sqrt(1.1);
173  u_upper *= sqrt(1.1);
174  while(u_lower - u_old - ratefact * CoolingRateFromU(u_lower, nHcgs, uvbg, ne_guess, Z) * dt > 0)
175  {
176  u_upper /= 1.1;
177  u_lower /= 1.1;
178  }
179  }
180 
181  do
182  {
183  u = 0.5 * (u_lower + u_upper);
184 
185  LambdaNet = CoolingRateFromU(u, nHcgs, uvbg, ne_guess, Z);
186 
187  if(u - u_old - ratefact * LambdaNet * dt > 0)
188  {
189  u_upper = u;
190  }
191  else
192  {
193  u_lower = u;
194  }
195 
196  du = u_upper - u_lower;
197 
198  iter++;
199 
200  if(iter >= (MAXITER - 10))
201  message(1, "u= %g\n", u);
202  }
203  while(fabs(du / u) > 1.0e-6 && iter < MAXITER);
204 
205  if(iter >= MAXITER)
206  {
207  endrun(10, "failed to converge in DoCooling()\n");
208  }
209 
210  u *= All.UnitDensity_in_cgs / All.UnitPressure_in_cgs; /* to internal units */
211 
212  return u;
213 }

274 static double solve_equilibrium_temp(double u, double nHcgs, struct UVBG * uvbg, struct abundance * y)
275 {
276  double temp, temp_old, temp_new, max = 0, ne_old;
277  double mu;
278  struct rates r;
279  int iter = 0;
280 /* double u_input, rho_input, ne_input; */
281 
282  mu = (1 + 4 * yhelium) / (1 + yhelium + y->ne);
283  temp = GAMMA_MINUS1 / BOLTZMANN * u * PROTONMASS * mu;
284 
285  do
286  {
287  ne_old = y->ne;
288 
289  find_abundances_and_rates(log10(temp), nHcgs, uvbg, y, &r);
290 
291  temp_old = temp;
292 
293  mu = (1 + 4 * yhelium) / (1 + yhelium + y->ne);
294 
295  temp_new = GAMMA_MINUS1 / BOLTZMANN * u * PROTONMASS * mu;
296 
297  max =
298  DMAX(max,
299  temp_new / (1 + yhelium + y->ne) * fabs((y->ne - ne_old) / (temp_new - temp_old + 1.0)));
300 
301  temp = temp_old + (temp_new - temp_old) / (1 + max);
302  iter++;
303 
304  if(iter > (MAXITER - 10))
305  message(1, "-> temp= %g ne=%g\n", temp, y->ne);
306  }
307  while(fabs(temp - temp_old) > 1.0e-3 * temp && iter < MAXITER);
308 
309  if(iter >= MAXITER)
310  {
311  endrun(12, "failed to converge in solve_equilibrium_temp()\n");
312  }
313 
314  return temp;
315 }

320 static void find_abundances_and_rates(double logT, double nHcgs, struct UVBG * uvbg, struct abundance * y, struct rates * r)
321 {
322  double neold, nenew;
323  int j, niter;
324  double flow, fhi, t;
325 
326 /* double logT_input, rho_input, ne_input; */
327 
328  if(logT <= Tmin) /* everything neutral */
329  {
330  memset(r, 0, sizeof(struct rates));
331  y->nH0 = 1.0;
332  y->nHe0 = yhelium;
333  y->nHp = 0;
334  y->nHep = 0;
335  y->nHepp = 0;
336  y->ne = 0;
337  return;
338  }
339 
340  if(logT >= Tmax) /* everything is ionized */
341  {
342  memset(r, 0, sizeof(struct rates));
343  y->nH0 = 0;
344  y->nHe0 = 0;
345  y->nHp = 1.0;
346  y->nHep = 0;
347  y->nHepp = yhelium;
348  y->ne = y->nHp + 2.0 * y->nHepp; /* note: in units of the hydrogen number density */
349  return;
350  }
351 
352  t = (logT - Tmin) / deltaT;
353  j = (int) t;
354  fhi = t - j;
355  flow = 1 - fhi;
356 
357  if(y->ne== 0)
358  y->ne = 1.0;
359 
360  niter = 0;
361  double necgs = y->ne * nHcgs;
362 
363  /* evaluate number densities iteratively (cf KWH eqns 33-38) in units of nH */
364  do
365  {
366  double gJH0ne, gJHe0ne, gJHepne;
367  niter++;
368 
369  r->aHp = flow * AlphaHp[j] + fhi * AlphaHp[j + 1];
370  r->aHep = flow * AlphaHep[j] + fhi * AlphaHep[j + 1];
371  r->aHepp = flow * AlphaHepp[j] + fhi * AlphaHepp[j + 1];
372  r->ad = flow * Alphad[j] + fhi * Alphad[j + 1];
373  r->geH0 = flow * GammaeH0[j] + fhi * GammaeH0[j + 1];
374  r->geHe0 = flow * GammaeHe0[j] + fhi * GammaeHe0[j + 1];
375  r->geHep = flow * GammaeHep[j] + fhi * GammaeHep[j + 1];
376 
377  if(necgs <= 1.e-25 || uvbg->J_UV == 0)
378  {
379  gJH0ne = gJHe0ne = gJHepne = 0;
380  }
381  else
382  {
383  gJH0ne = uvbg->gJH0 / necgs;
384  gJHe0ne = uvbg->gJHe0 / necgs;
385  gJHepne = uvbg->gJHep / necgs;
386  }
387 
388  y->nH0 = r->aHp / (r->aHp + r->geH0 + gJH0ne); /* eqn (33) */
389  y->nHp = 1.0 - y->nH0; /* eqn (34) */
390 
391  if((gJHe0ne + r->geHe0) <= SMALLNUM) /* no ionization at all */
392  {
393  y->nHep = 0.0;
394  y->nHepp = 0.0;
395  y->nHe0 = yhelium;
396  }
397  else
398  {
399  y->nHep = yhelium / (1.0 + (r->aHep + r->ad) / (r->geHe0 + gJHe0ne) + (r->geHep + gJHepne) / r->aHepp); /* eqn (35) */
400  y->nHe0 = y->nHep * (r->aHep + r->ad) / (r->geHe0 + gJHe0ne); /* eqn (36) */
401  y->nHepp = y->nHep * (r->geHep + gJHepne) / r->aHepp; /* eqn (37) */
402  }
403 
404  neold = y->ne;
405 
406  y->ne = y->nHp + y->nHep + 2 * y->nHepp; /* eqn (38) */
407 
408  if(uvbg->J_UV == 0)
409  break;
410 
411  nenew = 0.5 * (y->ne + neold);
412  y->ne = nenew;
413  necgs = y->ne * nHcgs;
414 
415  if(fabs(y->ne - neold) < 1.0e-4)
416  break;
417 
418  if(niter > (MAXITER - 10))
419  message(1, "ne= %g niter=%d\n", y->ne, niter);
420  }
421  while(niter < MAXITER);
422 
423  if(niter >= MAXITER)
424  {
425  endrun(13, "no convergence reached in find_abundances_and_rates()\n");
426  }
427 
428  r->bH0 = flow * BetaH0[j] + fhi * BetaH0[j + 1];
429  r->bHep = flow * BetaHep[j] + fhi * BetaHep[j + 1];
430  r->bff = flow * Betaff[j] + fhi * Betaff[j + 1];
431 }

499 /* Calculates (heating rate-cooling rate)/n_h^2 in cgs units
500 */
501 double PrimordialCoolingRate(double logT, double nHcgs, struct UVBG * uvbg, double *nelec)
502 {
503  double Lambda, Heat;
504  double redshift;
505  double T;
506  struct abundance y;
507  struct rates r;
508 
509  if(logT <= Tmin)
510  logT = Tmin + 0.5 * deltaT; /* floor at Tmin */
511 
512 
513  y.ne = *nelec;
514 
515  if(logT < Tmax)
516  {
517  double LambdaExc, LambdaIon, LambdaRec, LambdaFF;
518  double LambdaExcH0, LambdaExcHep, LambdaIonH0, LambdaIonHe0, LambdaIonHep;
519  double LambdaRecHp, LambdaRecHep, LambdaRecHepp, LambdaRecHepd;
520  find_abundances_and_rates(logT, nHcgs, uvbg, &y, &r);
521  *nelec = y.ne;
522  /* Compute cooling and heating rate (cf KWH Table 1) in units of nH**2 */
523  T = pow(10.0, logT);
524 
525  LambdaExcH0 = r.bH0 * y.ne * y.nH0;
526  LambdaExcHep = r.bHep * y.ne * y.nHep;
527  LambdaExc = LambdaExcH0 + LambdaExcHep; /* excitation */
528 
529  LambdaIonH0 = 2.18e-11 * r.geH0 * y.ne * y.nH0;
530  LambdaIonHe0 = 3.94e-11 * r.geHe0 * y.ne * y.nHe0;
531  LambdaIonHep = 8.72e-11 * r.geHep * y.ne * y.nHep;
532  LambdaIon = LambdaIonH0 + LambdaIonHe0 + LambdaIonHep; /* ionization */
533 
534  LambdaRecHp = 1.036e-16 * T * y.ne * (r.aHp * y.nHp);
535  LambdaRecHep = 1.036e-16 * T * y.ne * (r.aHep * y.nHep);
536  LambdaRecHepp = 1.036e-16 * T * y.ne * (r.aHepp * y.nHepp);
537  LambdaRecHepd = 6.526e-11 * r.ad * y.ne * y.nHep;
538  LambdaRec = LambdaRecHp + LambdaRecHep + LambdaRecHepp + LambdaRecHepd;
539 
540  LambdaFF = r.bff * (y.nHp + y.nHep + 4 * y.nHepp) * y.ne;
541 
542  Lambda = LambdaExc + LambdaIon + LambdaRec + LambdaFF;
543 
544  redshift = 1 / All.cf.a - 1;
545  double LambdaCmptn = 5.65e-36 * y.ne * (T - 2.73 * (1. + redshift)) * pow(1. + redshift, 4.) / nHcgs;
546 
547  Lambda += LambdaCmptn;
548 
549  Heat = 0;
550  double hefac=1.;
551  if(All.HeliumHeatOn)
552  hefac = he_reion_factor(nHcgs);
553  if(uvbg->J_UV != 0)
554  Heat += hefac * (y.nH0 * uvbg->epsH0 + y.nHe0 * uvbg->epsHe0 + y.nHep * uvbg->epsHep) / nHcgs;
555 
556  }
557  else /* here we're outside of tabulated rates, T>Tmax K */
558  {
559  /* at high T (fully ionized); only free-free and Compton cooling are present.
560  Assumes no heating. */
561  double LambdaFF, LambdaCmptn;
562  Heat = 0;
563 
564  /* very hot: H and He both fully ionized */
565  y.nHp = 1.0;
566  y.nHep = 0;
567  y.nHepp = yhelium;
568  y.ne = y.nHp + 2.0 * y.nHepp;
569  *nelec = y.ne; /* note: in units of the hydrogen number density */
570 
571  T = pow(10.0, logT);
572  LambdaFF =
573  1.42e-27 * sqrt(T) * (1.1 + 0.34 * exp(-(5.5 - logT) * (5.5 - logT) / 3)) * (y.nHp + 4 * y.nHepp) * y.ne;
574 
575  redshift = 1 / All.cf.a - 1;
576  /* add inverse Compton cooling off the microwave background */
577  LambdaCmptn = 5.65e-36 * y.ne * (T - 2.73 * (1. + redshift)) * pow(1. + redshift, 4.) / nHcgs;
578 
579  Lambda = LambdaFF + LambdaCmptn;
580  }
581 
582  return (Heat - Lambda);
583 }
