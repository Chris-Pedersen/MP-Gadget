311  return;
312  }

314  double r = iter->base.r;
315  double r2 = iter->base.r2;
316  double * dist = iter->base.dist;
317 
318 #ifdef SFR
319  if(All.WindOn && HAS(All.WindModel, WIND_DECOUPLE_SPH)) {
320  if(SPHP(other).DelayTime > 0) /* partner is a wind particle */
321  if(!(I->DelayTime > 0)) /* if I'm not wind, then ignore the wind particle */
322  return;
323  }
324 #endif

328  " We haven't implemented tracer particles and this shall not happen\n");

332  {
333

336  double dwk = density_kernel_dwk(&iter->kernel, u);

346

349  O->EgyRho += mass_j * EntPred * wk;

351 #endif

354  if(r > 0)
355  {
356  int d;
357  for (d = 0; d < 3; d ++) {
358  O->GradRho[d] += mass_j * dwk * dist[d] / r;
359  }
360  }
361 #endif

364  {
365  double fac = mass_j * dwk / r;
366  double dv[3];

379  }
380  }
