766 static int
767 domain_toptree_split(struct local_topnode_data * topTree, int * topTreeSize,
768  int i)
769 {
770  int j;
771  /* we ran out of top nodes and must get more.*/
772  if((*topTreeSize + 8) > MaxTopNodes)
773  return 1;
774 
775  if(topTree[i].Shift < 3) {
776  endrun(1, "Failed to build a TopTree -- particles overly clustered.\n");
777  }

779  /*Make a new topnode section attached to this node*/
780  topTree[i].Daughter = *topTreeSize;
781  (*topTreeSize) += 8;

784  for(j = 0; j < 8; j++)
785  {
786  const int sub = topTree[i].Daughter + j;

788  * and no daughters.*/
789  topTree[sub].Daughter = -1;
790  topTree[sub].Parent = i;
791  /* Shorten the peano key by a factor 8, reflecting the oct-tree level.*/
792  topTree[sub].Shift = topTree[i].Shift - 3;
793  /* This is the region of peanospace covered by this node.*/
794  topTree[sub].StartKey = topTree[i].StartKey + j * (1L << topTree[sub].Shift);
795  /* We will compute the cost and initialise the first particle in the node below.
796  * This PIndex value is never used*/
797  topTree[sub].PIndex = topTree[i].PIndex;
798  topTree[sub].Count = 0;
799  topTree[sub].Cost = 0;
800  }

888 /*
889  * This function performs local refinement of the topTree.
890  *
891  * It creates the local refinement by quickly going through
892  * a skeleton tree of a fraction (subsample) of all local particles.
893  *
894  * Next, we add flesh to the skeleton: all particles are deposited
895  * to the tree.
896  *
897  * Finally, in _truncation(), we chop off the cheap branches from the skeleton,
898  * terminating the branches when the cost and count are both sufficiently small.
899  *
900  * We do not use the full particle data. Sufficient to use LP, which contains
901  * the peano key and cost of each particle, and sorted by the peano key.
902  * */
903 static int
904 domain_check_for_local_refine_subsample(
905  DomainDecompositionPolicy * policy,
906  struct local_topnode_data * topTree, int * topTreeSize
907  )
908 {
909 
910  int i;
911 
912  struct local_particle_data * LP = (struct local_particle_data*) mymalloc("LocalParticleData", PartManager->NumPart * sizeof(LP[0]));
913 
914  /* Watchout : Peano/Morton ordering is required by the tree
915  * building algorithm in local_refine.
916  *
917  * We can either use a global or a local sorting here; the code will run
918  * without crashing.
919  *
920  * A global sorting is chosen to ensure the local topTrees are really local
921  * and the leaves almost disjoint. This makes the merged topTree a more accurate
922  * representation of the true cost / load distribution, for merging
923  * and secondary refinement are approximated.
924  *
925  * A local sorting may be faster but makes the tree less accurate due to
926  * more likely running into overlapped local topTrees.
927  * */
928 
929  int Nsample = PartManager->NumPart / policy->SubSampleDistance;
930 
931  if(Nsample == 0 && PartManager->NumPart != 0) Nsample = 1;
932 
933 #pragma omp parallel for
934  for(i = 0; i < PartManager->NumPart; i ++)
935  {
936  LP[i].Key = P[i].Key;
937  LP[i].Cost = domain_particle_costfactor(i);
938  }
939 
940  /* First sort to ensure spatially 'even' subsamples; FIXME: This can probably
941  * be omitted in most cases. Usually the particles in memory won't be very far off
942  * from a peano order. */
943  if(policy->PreSort)
944  qsort_openmp(LP, PartManager->NumPart, sizeof(struct local_particle_data), order_by_key);
945 
946  /* now subsample */
947  for(i = 0; i < Nsample; i ++)
948  {
949  LP[i].Key = LP[i * policy->SubSampleDistance].Key;
950  LP[i].Cost = LP[i * policy->SubSampleDistance].Cost;
951  }
952 
953  if(policy->UseGlobalSort) {
954  mpsort_mpi(LP, Nsample, sizeof(struct local_particle_data), mp_order_by_key, 8, NULL, MPI_COMM_WORLD);
955  } else {
956  qsort_openmp(LP, Nsample, sizeof(struct local_particle_data), order_by_key);
957  }
958 
959  walltime_measure("/Domain/DetermineTopTree/Sort");
960 
961  *topTreeSize = 1;
962  topTree[0].Daughter = -1;
963  topTree[0].Parent = -1;
964  topTree[0].Shift = BITS_PER_DIMENSION * 3;
965  topTree[0].StartKey = 0;
966  topTree[0].PIndex = 0;
967  topTree[0].Count = 0;
968  topTree[0].Cost = 0;
969 
970  /* The tree building algorithm here requires the LP structure to
971  * be sorted by key, in which case we either refine
972  * the node the last particle lives in, or jump into a new empty node,
973  * as the particles are scanned through.
974  *
975  * I unfortunately cannot find the direct reference of this with
976  * the proof. I found it around 2012~2013 when writing psphray2
977  * and didn't cite it properly then!
978  *
979  * Here is a blog link that is related:
980  *
981  * https://devblogs.nvidia.com/parallelforall/thinking-parallel-part-iii-tree-construction-gpu/
982  *
983  * A few interesting papers that may make this faster and cheaper.
984  *
985  * http://sc07.supercomputing.org/schedule/pdf/pap117.pdf
986  *
987  * */
988 
989  /* 1. create a skeleton of the toptree.
990  * We do not use all particles to avoid excessive memory consumption.
991  *
992  * */
993 
994 
995  /* During the loop, topTree[?].Count is a flag to indicate if
996  * the node has been finished. We either refine the last leaf node
997  * or create a new leaf because of the peano/morton sorting.
998  * */
999  peano_t last_key = -1;
1000  int last_leaf = -1;
1001  i = 0;
1002  while(i < Nsample) {
1003 
1004  int leaf = domain_toptree_get_subnode(topTree, LP[i].Key);
1005 
1006  if (leaf == last_leaf && topTree[leaf].Shift >= 3) {
1007  /* two particles in a node? need refinement if possible. */
1008  if(0 != domain_toptree_split(topTree, topTreeSize, leaf)) {
1009  /* out of memory, retry */
1010  return 1;
1011  }
1012  /* pop the last particle and reinsert it */
1013  topTree[leaf].Count = 0;
1014 
1015  last_leaf = domain_toptree_insert(topTree, last_key, 0);
1016  /* retry the current particle. */
1017  continue;
1018  } else {
1019  if(topTree[leaf].Count != 0) {
1020  /* meeting a node that's already been finished ?
1021  * This shall not happen when key is already sorted;
1022  * due to the sorting.
1023  */
1024  endrun(-1, "Failed to build the toptree\n");
1025  }
1026  /* this will create a new node. */
1027  last_key = LP[i].Key;
1028  last_leaf = domain_toptree_insert(topTree, last_key, 0);
1029  i += policy->SubSampleDistance;
1030  continue;
1031  }
1032  }
1033  /* Alternativly we could have kept last_cost in the above loop and avoid
1034  * the second and third step: */
1035 
1036  /* 2. Remove the skelton particles from the tree to make it a skeleton;
1037  * note that we never bothered to add Cost when the skeleton was built.
1038  * otherwise we shall clean it here too.*/
1039  for(i = 0; i < *topTreeSize; i ++ ) {
1040  topTree[i].Count = 0;
1041  }
1042 
1043  /* 3. insert all particles to the skeleton tree; Count will be correct because we cleaned them.*/
1044  for(i = 0; i < Nsample; i ++ ) {
1045  domain_toptree_insert(topTree, LP[i].Key, LP[i].Cost);
1046  }
1047 
1048  myfree(LP);
1049 
1050  /* then compute the costs of the internal nodes. */
1051  domain_toptree_update_cost(topTree, 0);
1052 
1053  /* we leave truncation in another function, for costlimit and countlimit must be
1054  * used in secondary refinement*/
1055  return 0;
1056 }

1058 int
1059 domain_nonrecursively_combine_topTree(struct local_topnode_data * topTree, int * topTreeSize)
1060 {
1061  /*
1062  * combine topTree non recursively, this uses MPI_Bcast within a group.
1063  * it shall be quite a bit faster (~ x2) than the old recursive scheme.
1064  *
1065  * it takes less time at higher sep.
1066  *
1067  * The communication should have been done with MPI Inter communicator.
1068  * but I couldn't figure out how to do it that way.
1069  * */
1070  int sep = 1;
1071  MPI_Datatype MPI_TYPE_TOPNODE;
1072  MPI_Type_contiguous(sizeof(struct local_topnode_data), MPI_BYTE, &MPI_TYPE_TOPNODE);
1073  MPI_Type_commit(&MPI_TYPE_TOPNODE);
1074  int errorflag = 0;
1075  int errorflagall = 0;
1076 
1077  for(sep = 1; sep < NTask; sep *=2) {
1078  /* build the subcommunicators for broadcasting */
1079  int Color = ThisTask / sep;
1080  int Key = ThisTask % sep;
1081  int ntopnodes_import = 0;
1082  struct local_topnode_data * topTree_import = NULL;
1083 
1084  int recvTask = -1; /* by default do not communicate */
1085 
1086  if(Key != 0) {
1087  /* non leaders will skip exchanges */
1088  goto loop_continue;
1089  }
1090 
1091  /* leaders of even color will combine nodes from next odd color,
1092  * so that when sep is increased eventually rank 0 will have all
1093  * nodes */
1094  if(Color % 2 == 0) {
1095  /* even guys recv */
1096  recvTask = ThisTask + sep;
1097  if(recvTask < NTask) {
1098  MPI_Recv(
1099  &ntopnodes_import, 1, MPI_INT, recvTask, TAG_GRAV_A,
1100  MPI_COMM_WORLD, MPI_STATUS_IGNORE);
1101  topTree_import = (struct local_topnode_data *) mymalloc("topTree_import",
1102  IMAX(ntopnodes_import, *topTreeSize) * sizeof(struct local_topnode_data));
1103 
1104  MPI_Recv(
1105  topTree_import,
1106  ntopnodes_import, MPI_TYPE_TOPNODE,
1107  recvTask, TAG_GRAV_B,
1108  MPI_COMM_WORLD, MPI_STATUS_IGNORE);
1109 
1110 
1111  if((*topTreeSize + ntopnodes_import) > MaxTopNodes) {
1112  errorflag = 1;
1113  } else {
1114  if(ntopnodes_import < 0) {
1115  endrun(1, "severe domain error using a unintended rank \n");
1116  }
1117  if(ntopnodes_import > 0 ) {
1118  domain_toptree_merge(topTree, topTree_import, 0, 0, topTreeSize);
1119  }
1120  }
1121  myfree(topTree_import);
1122  }
1123  } else {
1124  /* odd guys send */
1125  recvTask = ThisTask - sep;
1126  if(recvTask >= 0) {
1127  MPI_Send(topTreeSize, 1, MPI_INT, recvTask, TAG_GRAV_A,
1128  MPI_COMM_WORLD);
1129  MPI_Send(topTree,
1130  *topTreeSize, MPI_TYPE_TOPNODE,
1131  recvTask, TAG_GRAV_B,
1132  MPI_COMM_WORLD);
1133  }
1134  *topTreeSize = -1;
1135  }
1136 
1137 loop_continue:
1138  MPI_Allreduce(&errorflag, &errorflagall, 1, MPI_INT, MPI_LOR, MPI_COMM_WORLD);
1139  if(errorflagall) {
1140  break;
1141  }
1142  }
1143 
1144  MPI_Bcast(topTreeSize, 1, MPI_INT, 0, MPI_COMM_WORLD);
1145  MPI_Bcast(topTree, *topTreeSize, MPI_TYPE_TOPNODE, 0, MPI_COMM_WORLD);
1146  MPI_Type_free(&MPI_TYPE_TOPNODE);
1147  return errorflagall;
1148 }

1237 static int
1238 domain_global_refine(
1239  struct local_topnode_data * topTree, int * topTreeSize,
1240  int64_t countlimit, int64_t costlimit)
1241 {
1242  int i;
1243 
1244  /* At this point we have refined the local particle tree so that each
1245  * topNode contains a Cost and Count below the cost threshold. We have then
1246  * done a global merge of the particle tree. Some of our topTree may now contain
1247  * more particles than the Cost threshold, but repeating refinement using the local
1248  * algorithm is complicated - particles inside any particular topNode may be
1249  * on another processor. So we do a local volume based refinement here. This
1250  * just cuts each topNode above the threshold into 8 equal-sized portions by
1251  * subdividing the peano key.
1252  *
1253  * NOTE: Just like the merge, this does not correctly preserve costs!
1254  * Costs are just divided by 8, because recomputing them for the daughter nodes
1255  * will be expensive.
1256  * In practice this seems to work fine, probably because the cost distribution
1257  * is not that unbalanced. */
1258 
1259  message(0, "local topTree size before appending=%d\n", *topTreeSize);
1260 
1261  /*Note that *topTreeSize will change inside the loop*/
1262  for(i = 0; i < *topTreeSize; i++)
1263  {
1264  /*If this node has no children and non-zero size*/
1265  if(topTree[i].Daughter >= 0 || topTree[i].Shift <= 0) continue;
1266 
1267  /*If this node is also more costly than the limit*/
1268  if(topTree[i].Count < countlimit && topTree[i].Cost < costlimit) continue;
1269 
1270  /*If we have no space for another 8 topTree, exit */
1271  if((*topTreeSize + 8) > MaxTopNodes) {
1272  return 1;
1273  }
1274 
1275  topTree[i].Daughter = *topTreeSize;
1276  int j;
1277  for(j = 0; j < 8; j++)
1278  {
1279  int sub = topTree[i].Daughter + j;
1280  topTree[sub].Shift = topTree[i].Shift - 3;
1281  topTree[sub].Count = topTree[i].Count / 8;
1282  topTree[sub].Cost = topTree[i].Cost / 8;
1283  topTree[sub].Daughter = -1;
1284  topTree[sub].Parent = i;
1285  topTree[sub].StartKey = topTree[i].StartKey + j * (1L << topTree[sub].Shift);
1286  }
1287  (*topTreeSize) += 8;
1288  }
1289  return 0;
1290 }
