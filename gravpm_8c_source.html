379  * Store it in the PowerSpectrum structure */
380 void powerspectrum_add_mode(const int64_t k2, const int kpos[3], pfft_complex * const value, const double invwindow) {
381 
382  if(k2 == 0) {
383  /* Save zero mode corresponding to the mean as the normalisation factor.*/
384  PowerSpectrum.Norm = (value[0][0] * value[0][0] + value[0][1] * value[0][1]);
385  return;
386  }
387  /* Measure power spectrum: we don't want the zero mode.
388  * Some modes with k_z = 0 or N/2 have weight 1, the rest have weight 2.
389  * This is because of the symmetry of the real fft. */
390  if(k2 > 0) {
391  /*How many bins per unit (log) interval in k?*/
392  const double binsperunit=(PowerSpectrum.size-1)/log(sqrt(3)*All.Nmesh/2.0);
393  int kint=floor(binsperunit*log(k2)/2.);
394  int w;
395  const double keff = sqrt(kpos[0]*kpos[0]+kpos[1]*kpos[1]+kpos[2]*kpos[2]);
396  const double m = (value[0][0] * value[0][0] + value[0][1] * value[0][1]);
397  /*Make sure we do not overflow (although this should never happen)*/
398  if(kint >= PowerSpectrum.size)
399  return;
400  if(kpos[2] == 0 || kpos[2] == All.Nmesh/2) w = 1;
401  else w = 2;
402  /*Make sure we use thread-local memory to avoid racing.*/
403  const int index = kint + omp_get_thread_num() * PowerSpectrum.size;
404  /*Multiply P(k) by inverse window function*/
405  PowerSpectrum.Power[index] += w * m * invwindow * invwindow;
406  PowerSpectrum.Nmodes[index] += w;
407  PowerSpectrum.kk[index] += w * keff;
408  }

431 static void potential_transfer(int64_t k2, int kpos[3], pfft_complex *value) {
432 
433  const double asmth2 = pow((2 * M_PI) * All.Asmth / All.Nmesh,2);
434  double f = 1.0;
435  const double smth = exp(-k2 * asmth2) / k2;
436  /* the CIC deconvolution kernel is
437  *
438  * sinc_unnormed(k_x L / 2 All.Nmesh) ** 2
439  *
440  * k_x = kpos * 2pi / L
441  *
442  * */
443  int k;
444  for(k = 0; k < 3; k ++) {
445  double tmp = (kpos[k] * M_PI) / All.Nmesh;
446  tmp = sinc_unnormed(tmp);
447  f *= 1. / (tmp * tmp);
448  }
449  /*
450  * first decovolution is CIC in par->mesh
451  * second decovolution is correcting readout
452  * I don't understand the second yet!
453  * */
454  const double fac = pot_factor * smth * f * f;
455 
456  /*Add neutrino power if desired*/
457  if(All.MassiveNuLinRespOn && k2 > 0) {
458  /*Change the units of k to match those of logkk*/
459  double logk2 = log(sqrt(k2) * 2 * M_PI / (All.BoxSize * All.UnitLength_in_cm/ 3.085678e24 ));
460  /* Note get_neutrino_powerspec returns Omega_nu / (Omega0 -OmegaNu) * delta_nu / P_cdm^1/2, which is dimensionless.
461  * So below is: M_cdm * delta_cdm (1 + Omega_nu/(Omega0-OmegaNu) (delta_nu / delta_cdm))
462  * = M_cdm * (delta_cdm (Omega0 - OmegaNu)/Omega0 + Omega_nu/Omega0 delta_nu) * Omega0 / (Omega0-OmegaNu)
463  * = M_cdm * Omega0 / (Omega0-OmegaNu) * (delta_cdm (1 - f_nu) + f_nu delta_nu) )
464  * = M_cdm * Omega0 / (Omega0-OmegaNu) * delta_t
465  * = (M_cdm + M_nu) * delta_t
466  * This is correct for the forces, and gives the right power spectrum,
467  * once we multiply PowerSpectrum.Norm by (Omega0 / (Omega0 - OmegaNu))**2 */
468  const double nufac = 1 + get_dnudcdm_powerspec(&nu_pow, logk2);
469  value[0][0] *= nufac;
470  value[0][1] *= nufac;
471  }
472 
473  /*Compute the power spectrum*/
474  powerspectrum_add_mode(k2, kpos, value, f);
475  if(k2 == 0) {
476  if(All.MassiveNuLinRespOn) {
477  const double MtotbyMcdm = All.CP.Omega0/(All.CP.Omega0 - pow(All.Time,3)*get_omega_nu_nopart(&All.CP.ONu, All.Time));
478  PowerSpectrum.Norm *= MtotbyMcdm*MtotbyMcdm;
479  }
480  /* Remove zero mode corresponding to the mean.*/
481  value[0][0] = 0.0;
482  value[0][1] = 0.0;
483  return;
484  }
485 
486  value[0][0] *= fac;
487  value[0][1] *= fac;
488 }
