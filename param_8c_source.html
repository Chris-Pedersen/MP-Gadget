67 static int
68 OutputListAction(ParameterSet * ps, char * name, void * data)
69 {
70  char * outputlist = param_get_string(ps, name);
71  char * strtmp=strdup(outputlist);
72  char * token;
73  int count;
74 
75  /* Note TimeInit and TimeMax not yet initialised here*/
76 
77  /*First parse the string to get the number of outputs*/
78  for(count=0, token=strtok(strtmp,","); token; count++, token=strtok(NULL, ","))
79  {}
80 /* message(1, "Found %d times in output list.\n", count); */
81 
82  /*Allocate enough memory*/
83  All.OutputListLength = count;
84  int maxcount = DMAX(sizeof(All.OutputListTimes) / sizeof(All.OutputListTimes[0]), MAXSNAPSHOTS);
85  if(All.OutputListLength > maxcount) {
86  message(1, "Too many entries (%d) in the OutputList, can take no more than %d.\n", All.OutputListLength, maxcount);
87  return 1;
88  }
89  /*Now read in the values*/
90  for(count=0,token=strtok(outputlist,","); count < All.OutputListLength && token; count++, token=strtok(NULL,","))
91  {
92  /* Skip a leading quote if one exists.
93  * Extra characters are ignored by atof, so
94  * no need to skip matching char.*/
95  if(token[0] == '"')
96  token+=1;
97 
98  double a = atof(token);
99 
100  if(a < 0.0) {
101  endrun(1, "Requesting a negative output scaling factor a = %g\n", a);
102  }
103  All.OutputListTimes[count] = a;
104 /* message(1, "Output at: %g\n", All.OutputListTimes[count]); */
105  }
106  free(strtmp);
107  qsort(All.OutputListTimes, All.OutputListLength, sizeof(double), cmp_double);
108  return 0;
109 }
