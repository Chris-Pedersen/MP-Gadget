150 static void
151 sfr_wind_weight_postprocess(int i)
152 {
153  int diff = Wind[i].Ngb - 40;
154  if(diff < -2) {
155  /* too few */
156  Wind[i].Left = Wind[i].DMRadius;
157  } else if(diff > 2) {
158  /* too many */
159  Wind[i].Right = Wind[i].DMRadius;
160  } else {
161  P[i].DensityIterationDone = 1;
162  }
163  if(Wind[i].Right >= 0) {
164  /* if Ngb hasn't converged to 40, see if DMRadius converged*/
165  if(Wind[i].Right - Wind[i].Left < 1e-2) {
166  P[i].DensityIterationDone = 1;
167  } else {
168  Wind[i].DMRadius = 0.5 * (Wind[i].Left + Wind[i].Right);
169  }
170  } else {
171  Wind[i].DMRadius *= 1.3;
172  }
173 
174  if(P[i].DensityIterationDone) {
175  double vdisp = Wind[i].V2sum / Wind[i].Ngb;
176  int d;
177  for(d = 0; d < 3; d ++) {
178  Wind[i].Vmean[d] = Wind[i].V1sum[d] / Wind[i].Ngb;
179  vdisp -= Wind[i].Vmean[d] * Wind[i].Vmean[d];
180  }
181  Wind[i].Vdisp = sqrt(vdisp / 3);
182  } else {
183 #pragma omp atomic
184  NPLeft ++;
185  }
186 }

196 static void
197 sfr_cool_postprocess(int i, TreeWalk * tw)
198 {
199  int flag;
200  /*Remove a wind particle from the delay mode if the (physical) density has dropped sufficiently.*/
201  if(SPHP(i).DelayTime > 0 && SPHP(i).Density * All.cf.a3inv < All.WindFreeTravelDensFac * All.PhysDensThresh) {
202  SPHP(i).DelayTime = 0;
203  }
204  /*Reduce the time until the particle can form stars again by the current timestep*/
205  if(SPHP(i).DelayTime > 0) {
206  const double dloga = get_dloga_for_bin(P[i].TimeBin);
207  /* the proper time duration of the step */
208  const double dtime = dloga / All.cf.hubble;
209  SPHP(i).DelayTime = DMAX(SPHP(i).DelayTime - dtime, 0);
210  }
211 
212  /* check whether conditions for star formation are fulfilled.
213  *
214  * f=1 normal cooling
215  * f=0 star formation
216  */
217  flag = get_sfr_condition(i);
218 
219  /* normal implicit isochoric cooling */
220  if(flag == 1 || All.QuickLymanAlphaProbability > 0) {
221  cooling_direct(i);
222  }
223  if(flag == 0) {
224  /* active star formation */
225  starformation(i);
226  }

369 static void
370 cooling_direct(int i) {
371 
372  /* the actual time-step */
373  double dloga = get_dloga_for_bin(P[i].TimeBin);
374  double dtime = dloga / All.cf.hubble;
375 
376  double ne = SPHP(i).Ne; /* electron abundance (gives ionization state and mean molecular weight) */
377 
378  double unew = DMAX(All.MinEgySpec,
379  (SPHP(i).Entropy + SPHP(i).DtEntropy * dloga) /
380  GAMMA_MINUS1 * pow(SPHP(i).EOMDensity * All.cf.a3inv, GAMMA_MINUS1));
381 
382 #ifdef BLACK_HOLES
383  if(SPHP(i).Injected_BH_Energy)
384  {
385  if(P[i].Mass == 0) {
386  endrun(-1, "Encoutered zero mass particle during sfr;"
387  " We haven't implemented tracer particles and this shall not happen\n");
388  /* This shall not happend */
389  SPHP(i).Injected_BH_Energy = 0;
390  }
391 
392  unew += SPHP(i).Injected_BH_Energy / P[i].Mass;
393  double temp = u_to_temp_fac * unew;
394 
395 
396  if(temp > 5.0e9)
397  unew = 5.0e9 / u_to_temp_fac;
398 
399  SPHP(i).Injected_BH_Energy = 0;
400  }
401 #endif
402 
403  struct UVBG uvbg;
404  GetParticleUVBG(i, &uvbg);
405  unew = DoCooling(unew, SPHP(i).Density * All.cf.a3inv, dtime, &uvbg, &ne, SPHP(i).Metallicity);
406 
407  SPHP(i).Ne = ne;
408 
409  if(P[i].TimeBin) /* upon start-up, we need to protect against dt==0 */
410  {
411  /* note: the adiabatic rate has been already added in ! */
412 
413  if(dloga > 0)
414  {
415 
416  SPHP(i).DtEntropy = (unew * GAMMA_MINUS1 /
417  pow(SPHP(i).EOMDensity * All.cf.a3inv,
418  GAMMA_MINUS1) - SPHP(i).Entropy) / dloga;
419 
420  if(SPHP(i).DtEntropy < -0.5 * SPHP(i).Entropy / dloga)
421  SPHP(i).DtEntropy = -0.5 * SPHP(i).Entropy / dloga;
422  }
423  }
424 }

429 /* returns 0 if the particle is actively forming stars */
430 static int get_sfr_condition(int i) {
431  int flag = 1;
432 /* no sfr !*/
433  if(!All.StarformationOn) {
434  return flag;
435  }
436  if(SPHP(i).Density * All.cf.a3inv >= All.PhysDensThresh)
437  flag = 0;
438 
439  if(SPHP(i).Density < All.OverDensThresh)
440  flag = 1;
441 
442  /* massless particles never form stars! */
443  if(P[i].Mass == 0) {
444  endrun(-1, "Encoutered zero mass particle during sfr ;"
445  " We haven't implemented tracer particles and this shall not happen\n");
446  }
447 
448  if(SPHP(i).DelayTime > 0)
449  flag = 1; /* only normal cooling for particles in the wind */
450 
451  if(All.QuickLymanAlphaProbability > 0) {
452  double dloga = get_dloga_for_bin(P[i].TimeBin);
453  double unew = DMAX(All.MinEgySpec,
454  (SPHP(i).Entropy + SPHP(i).DtEntropy * dloga) /
455  GAMMA_MINUS1 * pow(SPHP(i).EOMDensity * All.cf.a3inv, GAMMA_MINUS1));
456 
457  double temp = u_to_temp_fac * unew;
458 
459  if(SPHP(i).Density > All.OverDensThresh && temp < 1.0e5)
460  flag = 0;
461  else
462  flag = 1;
463  }
464 
465  return flag;
466 }

525 static void
526 sfr_wind_weight_ngbiter(TreeWalkQueryWind * I,
527  TreeWalkResultWind * O,
528  TreeWalkNgbIterWind * iter,
529  LocalTreeWalk * lv)
530 {
531  /* this evaluator walks the tree and sums the total mass of surrounding gas
532  * particles as described in VS08. */
533  /* it also calculates the DM dispersion of the nearest 40 DM paritlces */
534  if(iter->base.other == -1) {
535  double hsearch = DMAX(I->Hsml, I->DMRadius);
536  iter->base.Hsml = hsearch;
537  iter->base.mask = 1 + 2; /* gas and dm */
538  iter->base.symmetric = NGB_TREEFIND_ASYMMETRIC;
539  return;
540  }
541 
542  int other = iter->base.other;
543  double r = iter->base.r;
544  double * dist = iter->base.dist;
545 
546  if(P[other].Type == 0) {
547  if(r > I->Hsml) return;
548  /* Ignore wind particles */
549  if(SPHP(other).DelayTime > 0) return;
550  /* NOTE: think twice if we want a symmetric tree walk when wk is used. */
551  //double wk = density_kernel_wk(&kernel, r);
552  double wk = 1.0;
553  O->TotalWeight += wk * P[other].Mass;
554  }
555 
556  if(P[other].Type == 1) {
557  if(r > I->DMRadius) return;
558  O->Ngb ++;
559  int d;
560  for(d = 0; d < 3; d ++) {
561  /* Add hubble flow; FIXME: this shall be a function, and the direction looks wrong too. */
562  double vel = P[other].Vel[d] + All.cf.hubble * All.cf.a * All.cf.a * dist[d];
563  O->V1sum[d] += vel;
564  O->V2sum += vel * vel;
565  }
566  }
567 
568  /*
569  message(1, "ThisTask = %d %ld ngb=%d NGB=%d TotalWeight=%g V2sum=%g V1sum=%g %g %g\n",
570  ThisTask, I->ID, numngb, O->Ngb, O->TotalWeight, O->V2sum,
571  O->V1sum[0], O->V1sum[1], O->V1sum[2]);
572  */
573 }

575 static void
576 sfr_wind_feedback_ngbiter(TreeWalkQueryWind * I,
577  TreeWalkResultWind * O,
578  TreeWalkNgbIterWind * iter,
579  LocalTreeWalk * lv)
580 {
581 
582  /* this evaluator walks the tree and blows wind. */
583 
584  if(iter->base.other == -1) {
585  iter->base.mask = 1;
586  iter->base.symmetric = NGB_TREEFIND_ASYMMETRIC;
587  iter->base.Hsml = I->Hsml;
588  return;
589  }
590  int other = iter->base.other;
591  double r = iter->base.r;
592 
593  /* skip wind particles */
594  if(SPHP(other).DelayTime > 0) return;
595 
596  /* this is radius cut is redundant because the tree walk is asymmetric
597  * we may want to use fancier weighting that requires symmetric in the future. */
598  if(r > I->Hsml) return;
599 
600  double windeff=0;
601  double v=0;
602  if(HAS(All.WindModel, WIND_FIXED_EFFICIENCY)) {
603  windeff = All.WindEfficiency;
604  v = All.WindSpeed * All.cf.a;
605  } else if(HAS(All.WindModel, WIND_USE_HALO)) {
606  windeff = 1.0 / (I->Vdisp / All.cf.a / All.WindSigma0);
607  windeff *= windeff;
608  v = All.WindSpeedFactor * I->Vdisp;
609  } else {
610  endrun(1, "WindModel = 0x%X is strange. This shall not happen.\n", All.WindModel);
611  }
612 
613  //double wk = density_kernel_wk(&kernel, r);
614 
615  /* in this case the particle is already locked by the tree walker */
616  /* we may want to add another lock to avoid this. */
617  if(P[other].ID != I->base.ID)
618  lock_particle(other);
619 
620  double wk = 1.0;
621  double p = windeff * wk * I->Mass / I->TotalWeight;
622  double random = get_random_number(I->base.ID + P[other].ID);
623  if (random < p) {
624  make_particle_wind(I->base.ID, other, v, I->Vmean);
625  }
626 
627  if(P[other].ID != I->base.ID)
628  unlock_particle(other);
629 
630 }

632 static int make_particle_wind(MyIDType ID, int i, double v, double vmean[3]) {

634  /* returns 0 if particle i is converted to wind. */

783

812 static double get_starformation_rate_full(int i, double dtime, MyFloat * ne_new, double * trelax, double * egyeff) {
813  double rateOfSF;
814  int flag;
815  double tsfr;
816  double factorEVP, egyhot, ne, tcool, y, x, cloudmass;
817  struct UVBG uvbg;
818 
819  flag = get_sfr_condition(i);
820 
821  if(flag == 1) {
822  /* this shall not happen but let's put in some safe
823  * numbers in case the code run wary!
824  *
825  * the only case trelax and egyeff are
826  * required is in starformation(i)
827  * */
828  if (trelax) {
829  *trelax = All.MaxSfrTimescale;
830  }
831  if (egyeff) {
832  *egyeff = All.EgySpecCold;
833  }
834  return 0;
835  }
836 
837  tsfr = sqrt(All.PhysDensThresh / (SPHP(i).Density * All.cf.a3inv)) * All.MaxSfrTimescale;
838  /*
839  * gadget-p doesn't have this cap.
840  * without the cap sm can be bigger than cloudmass.
841  */
842  if(tsfr < dtime)
843  tsfr = dtime;
844 
845  GetParticleUVBG(i, &uvbg);
846 
847  factorEVP = pow(SPHP(i).Density * All.cf.a3inv / All.PhysDensThresh, -0.8) * All.FactorEVP;
848 
849  egyhot = All.EgySpecSN / (1 + factorEVP) + All.EgySpecCold;
850 
851  ne = SPHP(i).Ne;
852 
853  tcool = GetCoolingTime(egyhot, SPHP(i).Density * All.cf.a3inv, &uvbg, &ne, SPHP(i).Metallicity);
854  y = tsfr / tcool * egyhot / (All.FactorSN * All.EgySpecSN - (1 - All.FactorSN) * All.EgySpecCold);
855 
856  x = 1 + 1 / (2 * y) - sqrt(1 / y + 1 / (4 * y * y));
857 
858  cloudmass = x * P[i].Mass;
859 
860  rateOfSF = (1 - All.FactorSN) * cloudmass / tsfr;
861 
862  if (ne_new ) {
863  *ne_new = ne;
864  }
865 
866  if (trelax) {
867  *trelax = tsfr * (1 - x) / x / (All.FactorSN * (1 + factorEVP));
868  }
869  if (egyeff) {
870  *egyeff = egyhot * (1 - x) + All.EgySpecCold * x;
871  }
872 
873  if (HAS(All.StarformationCriterion, SFR_CRITERION_MOLECULAR_H2)) {
874  rateOfSF *= get_sfr_factor_due_to_h2(i);
875  }
876  if (HAS(All.StarformationCriterion, SFR_CRITERION_SELFGRAVITY)) {
877  rateOfSF *= get_sfr_factor_due_to_selfgravity(i);
878  }
879  return rateOfSF;
880 }
