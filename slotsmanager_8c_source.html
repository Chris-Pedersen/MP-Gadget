144 static int
145 slots_gc_compact(int used, int ptype, size_t size)
146 {
147  /*Find first garbage particle*/
148  int i, nextgc = used;
149  for(i = 0; i < used; i++)
150  if(GARBAGE(i,ptype)) {
151  nextgc = i;
152  break;
153  }
154 
155  int ngc = 0;
156  /*Note each particle is tested exactly once*/
157  while(nextgc < used) {
158  int i;
159  /*Now lastgc contains a garbage*/
160  int lastgc = nextgc;
161  /*Find a non-garbage after it*/
162  int src = used;
163  for(i = lastgc + 1; i < used; i++)
164  if(!GARBAGE(i, ptype)) {
165  src = i;
166  break;
167  }
168  /*If no more non-garbage particles, don't both copying, just add a skip*/
169  if(src == used) {
170  ngc += src - lastgc;
171  break;
172  }
173  /*Destination is shifted already*/
174  int dest = lastgc - ngc;
175 
176  nextgc = used;
177  /*Find another garbage particle*/
178  for(i = src+1; i < used; i++)
179  if(GARBAGE(i, ptype)) {
180  nextgc = i;
181  break;
182  }
183  /*Add number of particles we skipped*/
184  ngc += src - lastgc;
185  int nmove = nextgc - src +1;
186 // message(1,"i = %d, PI = %d-> %d, nm=%d\n",i, src, dest, nmove);
187  memmove(PART(dest, ptype),PART(src, ptype),nmove*size);
188  }
