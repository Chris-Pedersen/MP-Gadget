382 double
383 get_timestep_dloga(const int p)
384 {
385  double ac = 0;
386  double dt = 0, dt_courant = 0;
387 
388  /*Compute physical acceleration*/
389  {
390  double ax = All.cf.a2inv * P[p].GravAccel[0];
391  double ay = All.cf.a2inv * P[p].GravAccel[1];
392  double az = All.cf.a2inv * P[p].GravAccel[2];
393 
394  ax += All.cf.a2inv * P[p].GravPM[0];
395  ay += All.cf.a2inv * P[p].GravPM[1];
396  az += All.cf.a2inv * P[p].GravPM[2];
397 
398  if(P[p].Type == 0)
399  {
400  const double fac2 = 1 / pow(All.Time, 3 * GAMMA - 2);
401  ax += fac2 * SPHP(p).HydroAccel[0];
402  ay += fac2 * SPHP(p).HydroAccel[1];
403  az += fac2 * SPHP(p).HydroAccel[2];
404  }
405 
406  ac = sqrt(ax * ax + ay * ay + az * az); /* this is now the physical acceleration */
407  }
408 
409  if(ac == 0)
410  ac = 1.0e-30;
411 
412  /* mind the factor 2.8 difference between gravity and softening used here. */
413  dt = sqrt(2 * All.ErrTolIntAccuracy * All.cf.a * (FORCE_SOFTENING(p) / 2.8) / ac);
414 
415  if(P[p].Type == 0)
416  {
417  const double fac3 = pow(All.Time, 3 * (1 - GAMMA) / 2.0);
418  dt_courant = 2 * All.CourantFac * All.Time * P[p].Hsml / (fac3 * SPHP(p).MaxSignalVel);
419  if(dt_courant < dt)
420  dt = dt_courant;
421  }
422 
423 #ifdef BLACK_HOLES
424  if(P[p].Type == 5)
425  {
426  if(BHP(p).Mdot > 0 && BHP(p).Mass > 0)
427  {
428  double dt_accr = 0.25 * BHP(p).Mass / BHP(p).Mdot;
429  if(dt_accr < dt)
430  dt = dt_accr;
431  }
432  if(BHP(p).TimeBinLimit > 0) {
433  double dt_limiter = get_dloga_for_bin(BHP(p).TimeBinLimit) / All.cf.hubble;
434  if (dt_limiter < dt) dt = dt_limiter;
435  }
436  }
437 #endif
438 
439  /* d a / a = dt * H */
440  double dloga = dt * All.cf.hubble;
441 
442  return dloga;
443 }

507  int count[6];
508  int64_t count_sum[6];
509  double v[6], v_sum[6], mim[6], min_mass[6];
510  double dloga = All.MaxSizeTimestep;
511 
512  for(type = 0; type < 6; type++)
513  {
514  count[type] = 0;
515  v[type] = 0;
516  mim[type] = 1.0e30;
517  }
518

520  {
521  v[P[i].Type] += P[i].Vel[0] * P[i].Vel[0] + P[i].Vel[1] * P[i].Vel[1] + P[i].Vel[2] * P[i].Vel[2];
522  if(P[i].Mass > 0)
523  {
524  if(mim[P[i].Type] > P[i].Mass)
525  mim[P[i].Type] = P[i].Mass;
526  }
527  count[P[i].Type]++;
528  }
529 
530  MPI_Allreduce(v, v_sum, 6, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
531  MPI_Allreduce(mim, min_mass, 6, MPI_DOUBLE, MPI_MIN, MPI_COMM_WORLD);
532 
533  sumup_large_ints(6, count, count_sum);
534 
535 #ifdef SFR
536  /* add star and gas particles together to treat them on equal footing, using the original gas particle
537  spacing. */
538  v_sum[0] += v_sum[4];
539  count_sum[0] += count_sum[4];
540  v_sum[4] = v_sum[0];
541  count_sum[4] = count_sum[0];
542 #ifdef BLACK_HOLES
543  v_sum[0] += v_sum[5];
544  count_sum[0] += count_sum[5];
545  v_sum[5] = v_sum[0];
546  count_sum[5] = count_sum[0];
547  min_mass[5] = min_mass[0];
548 #endif
549 #endif
550 
551  for(type = 0; type < 6; type++)
552  {
553  if(count_sum[type] > 0)
554  {
555  double omega, dmean, dloga1;
556  const double asmth = All.Asmth * All.BoxSize / All.Nmesh;
557  if(type == 0 || (type == 4 && All.StarformationOn)
558 #ifdef BLACK_HOLES
559  || (type == 5)
560 #endif
561  ) {
562  omega = All.CP.OmegaBaryon;
563  }
564  /* In practice usually FastParticleType == 2

566  else if (type == 2) {
567  omega = get_omega_nu(&All.CP.ONu, 1);
568  } else {
569  omega = All.CP.OmegaCDM;
570  }
571  /* "Avg. radius" of smallest particle: (min_mass/total_mass)^1/3 */
572  dmean = pow(min_mass[type] / (omega * 3 * All.CP.Hubble * All.CP.Hubble / (8 * M_PI * All.G)), 1.0 / 3);
573 
574  dloga1 = All.MaxRMSDisplacementFac * All.cf.hubble * All.cf.a * All.cf.a * DMIN(asmth, dmean) / sqrt(v_sum[type] / count_sum[type]);

576  type, dmean, asmth, min_mass[type], All.Time, sqrt(v_sum[type] / count_sum[type]), dloga);
577 
578  /* don't constrain the step to the neutrinos */
579  if(type != All.FastParticleType && dloga1 < dloga)
580  dloga = dloga1;
581  }
582  }
583  return dloga;
584 }

620 /* This function reverse the direction of the gravitational force.
621  * This is only useful for making Lagrangian glass files*/
622 void reverse_and_apply_gravity()
623 {
624  double dispmax=0, globmax;
625  int i;
626  for(i = 0; i < PartManager->NumPart; i++)
627  {
628  int j;
629  /*Reverse the direction of acceleration*/
630  for(j = 0; j < 3; j++)
631  {
632  P[i].GravAccel[j] *= -1;
633  P[i].GravAccel[j] -= P[i].GravPM[j];
634  P[i].GravPM[j] = 0;
635  }
636 
637  double disp = sqrt(P[i].GravAccel[0] * P[i].GravAccel[0] +
638  P[i].GravAccel[1] * P[i].GravAccel[1] + P[i].GravAccel[2] * P[i].GravAccel[2]);
639 
640  disp *= 2.0 / (3 * All.CP.Hubble * All.CP.Hubble);
641 
642  if(disp > dispmax)
643  dispmax = disp;
644  }
645 
646  MPI_Allreduce(&dispmax, &globmax, 1, MPI_DOUBLE, MPI_MAX, MPI_COMM_WORLD);
647 
648  double dmean = pow(P[0].Mass / (All.CP.Omega0 * 3 * All.CP.Hubble * All.CP.Hubble / (8 * M_PI * All.G)), 1.0 / 3);
649 
650  const double fac = DMIN(1.0, dmean / globmax);
651 
652  message(0, "Glass-making: dmean= %g global disp-maximum= %g\n", dmean, globmax);
653 
654  /* Move the actual particles according to the (reversed) gravitational force.
655  * Not sure why this is here rather than in the main code.*/
656  for(i = 0; i < PartManager->NumPart; i++)
657  {
658  int j;
659  for(j = 0; j < 3; j++)
660  {
661  P[i].Vel[j] = 0;
662  P[i].Pos[j] += fac * P[i].GravAccel[j] * 2.0 / (3 * All.CP.Hubble * All.CP.Hubble);
663  P[i].GravAccel[j] = 0;
664  }
665  }
666 
667 }
